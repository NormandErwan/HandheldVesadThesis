\newcommand{\figureETS}[3]{
	\begin{figure}
		\centering % Les figures doivent être centrées
		\fbox{ % Les figures doivent être délimitées par un rectangle
			\includegraphics[width=0.75\textwidth]{#1} % Spécification du paramètres "width", la largeur de l'image (en conservant les proportions). La largeur est ici restreinte à 0.75 fois "\textwidth", qui est la largeur maximale du texte dans le document en prenant en compte les marges
		}
		 \\ \parbox{0.75\textwidth}{\caption{#2}\label{fig:#3}} % Utilisation d'une parbox pour restreindre la largeur de la légende. Ici la taille maximale a été fixée à la largeur choisie pour l'image (0.75\textwidth). Le décanat demande d'éviter d'avoir des légendes qui dépasse les figures, dans la mesure du possible (si l'image est trop petite, la légende peut dépasser sa largeur).
	\end{figure}
}
